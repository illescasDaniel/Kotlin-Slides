	\begin{frame}[fragile]{Qué es}	
		\begin{itemize}
			\normalSizeItem {
				\textbf{Kotlin} es un lenguaje de propósito general, estáticamente tipado, que combina orientación a objetos con programación funcional.
			}
			\normalSizeItem {
				Funciona donde \textbf{Java} funcione: apps móviles, aplicaciones de escritorio, apps del lado del servidor.
			}
			\normalSizeItem {
				Tiene \textbf{soporte} oficial para:
				\begin{itemize}
					\normalSizeItem { 
						\textbf{IntelliJ IDEA} y Android Studio 
					}
					\normalSizeItem { Eclipse }
					\normalSizeItem { Maven, Gradle, Ant }
					\normalSizeItem { Spring Boot }
					\normalSizeItem { etc }
					\smallCite{KotlinJetBrainsBlog}
				\end{itemize}
			}
		\end{itemize}
	\end{frame}
		 
	\begin{frame}[fragile]{Quién está detrás}	
		\begin{itemize}
			\normalSizeItem {
				Es un lenguaje \textbf{open-source}, disponible en GitHub y con más de 250 contribuyentes.
			}
			\normalSizeItem {
				\textbf{JetBrains} es el desarrollador y principal patrocinador de Kotlin.
			}
		\end{itemize}
	\end{frame}
